def change(department, what_change, change)
  department.send("#{what_change}=", change)
  Database.instance.change_node({ 'table' => 'Departments', 'what_change' => 'DepartmentName', 'change' => change,
                                  'field' => 'DepartmentID', 'id' => department.id })
end
